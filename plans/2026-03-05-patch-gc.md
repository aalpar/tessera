# Patch GC Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Clean up flattened patches and stale block references after compaction.

**Architecture:** `RemovePatches` removes specific patch entries from PatchIndex (same add-wins pattern as `BlockRef.RemoveRef`). `CompactFile` orchestrates the full cleanup: flatten, remove patches, remove old block refs. No dominance check needed — the CRDT algebra guarantees concurrent adds survive removal.

**Tech Stack:** Go stdlib + `github.com/aalpar/crdt` (ormap, dotcontext). No new deps.

**Design doc:** `docs/plans/2026-03-05-patch-gc-design.md`

// RemovePatches removes specific patch entries for a file and returns a delta.
// Concurrent adds from other replicas survive (add-wins semantics).
func (p *PatchIndex) RemovePatches(fileID string, entries []patchEntry) *PatchIndex {
	delta := p.inner.Apply(fileID, func(id dotcontext.ReplicaID, ctx *dotcontext.CausalContext, v *dotcontext.DotMap[patchEntry, *dotcontext.DotSet], delta *dotcontext.DotMap[patchEntry, *dotcontext.DotSet]) {
		for _, entry := range entries {
			v.Delete(entry)
		}
	})
	return &PatchIndex{inner: delta}
}
```

This follows the `BlockRef.RemoveRef` pattern: we delete entries from the local DotMap inside `ORMap.Apply`. The delta carries the observed dots in its context but an empty (or reduced) store, so concurrent adds with new dots survive.

package tessera

import (
	"context"
	"fmt"
)

// CompactDeltas groups the deltas produced by CompactFile for replication.
type CompactDeltas struct {
	BlockRefAdds    []*BlockRef  // new chunks registered
	BlockRefRemoves []*BlockRef  // old chunks deregistered
	PatchRemove     *PatchIndex  // flattened patches removed
}

// CompactFile flattens a patched file into a clean recipe, removes the
// flattened patches from the PatchIndex, and deregisters old block refs.
//
// No dominance check is needed — the CRDT algebra guarantees that
// concurrent patch adds from other replicas survive the removal.
func CompactFile(
	ctx context.Context,
	fileID string,
	recipe *SnapshotRecipe,
	store BlockStore,
	index *BlockRef,
	patches *PatchIndex,
	chunker *Chunker,
) (*SnapshotRecipe, *CompactDeltas, error) {
	// 1. Snapshot patches before flatten
	currentPatches := patches.Patches(fileID)

	// 2. Flatten: read patched data, re-chunk, store new blocks
	newRecipe, addDeltas, err := Flatten(ctx, fileID, recipe, store, index, patches, chunker)
	if err != nil {
		return nil, nil, fmt.Errorf("compact %s: %w", fileID, err)
	}

	// 3. Remove flattened patches from PatchIndex
	var patchDelta *PatchIndex
	if len(currentPatches) > 0 {
		patchDelta = patches.RemovePatches(fileID, currentPatches)
	}

	// 4. Remove old block refs for old recipe's chunks
	oldHashes := make([]string, len(recipe.Blocks))
	for i, b := range recipe.Blocks {
		oldHashes[i] = b.Hash
	}
	// Also remove the old recipe block itself
	oldHashes = append(oldHashes, recipe.Version)
	removeDeltas := index.RemoveFileRefs(fileID, oldHashes)

	return newRecipe, &CompactDeltas{
		BlockRefAdds:    addDeltas,
		BlockRefRemoves: removeDeltas,
		PatchRemove:     patchDelta,
	}, nil
}
```

func TestCompactFileConcurrentPatchSurvives(t *testing.T) {
	store, _ := NewFSBlockStore(t.TempDir())
	index := New("w1")
	patches1 := NewPatchIndex("w1")
	patches2 := NewPatchIndex("w2")
	chunker := NewChunker(DefaultChunkerConfig())
	ctx := context.Background()

	// Write original file
	data := make([]byte, 30*1024)
	rand.Read(data)
	recipe, _, err := WriteSnapshot(ctx, "file-a", bytes.NewReader(data), chunker, store, index)
	if err != nil {
		t.Fatal(err)
	}

	// Worker 1 adds P1, syncs to worker 2
	_, err = WritePatch(ctx, "w1", "file-a", 5000, []byte("PATCH_1"), store, patches1)
	if err != nil {
		t.Fatal(err)
	}
	// Sync P1 to patches2 (patches2 needs to see it so compact can work)
	// For simplicity, we use the same store, and just need the PatchIndex synced
	_, err = WritePatch(ctx, "w2", "file-a", 5000, []byte("PATCH_1"), store, patches2)
	if err != nil {
		t.Fatal(err)
	}

	// Worker 1 compacts — this removes P1 from patches1
	newRecipe, deltas, err := CompactFile(ctx, "file-a", recipe, store, index, patches1, chunker)
	if err != nil {
		t.Fatal(err)
	}

	// Worker 2 concurrently adds P2 (hasn't seen the compact yet)
	p2Data := []byte("CONCURRENT_PATCH")
	dAdd, err := WritePatch(ctx, "w2", "file-a", 10000, p2Data, store, patches2)
	if err != nil {
		t.Fatal(err)
	}

	// Sync: worker 1 gets P2, worker 2 gets compact's patch removal
	patches1.Merge(dAdd)
	if deltas.PatchRemove != nil {
		patches2.Merge(deltas.PatchRemove)
	}

	// Worker 1 should see P2 surviving
	p1Patches := patches1.Patches("file-a")
	if len(p1Patches) != 1 {
		t.Fatalf("w1: expected 1 surviving patch, got %d", len(p1Patches))
	}

	// Worker 2 should also see only P2
	p2Patches := patches2.Patches("file-a")
	if len(p2Patches) != 1 {
		t.Fatalf("w2: expected 1 surviving patch, got %d", len(p2Patches))
	}

	// P2 should apply correctly over the new recipe
	got, err := PatchedReadRange(ctx, newRecipe, store, patches1, "file-a", 10000, uint64(len(p2Data)))
	if err != nil {
		t.Fatal(err)
	}
	if !bytes.Equal(got, p2Data) {
		t.Fatal("concurrent patch should apply correctly over compacted recipe")
	}
}
